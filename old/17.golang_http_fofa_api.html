上面列出了几个EX，我们注意到每个结束都调用了Close()，但是http和tcp不一样，又是无状态的协议，应该不用关闭吧。

首先我们来分析一下问题：http和tcp的close是否是同一个含义？因为http在tcp的上层。

Google关键字：golang http close

我Google到了这个：https://segmentfault.com/a/1190000020086816

综上分析，可以发现，readLoop和 writeLoop 两个goroutine在 写入请求并获取response返回后，并没有跳出for循环，而继续阻塞在 下一次for循环的select 语句里面，所以，两个函数所在的goroutine并没有运行结束，导致了最开的现象: goroutine持续增加导致内存持续增加

Go应该是对同一个url的并发做出了优化，不会跳出for循环，但是多次访问不同的url就会导致内存泄漏了。可以看出来，http的close和tcp压根就不是一回事！哈哈，没事，我们随手关闭一下就行。

package shodan
​
import (
    "encoding/json"
    "fmt"
    "net/http"
)
​
type APIInfo struct {
    QueryCredits int    `json:"query_credits"`
    ScanCredits  int    `json:"scan_credits"`
    Telnet       bool   `json:"telnet"`
    Plan         string `json:"plan"`
    Https        bool   `json:"https"`
    Unlocked     bool   `json:"unlocked"`
}//定义一个结构体，利用结构体标签来解析返回的json
​
func (s *Client) APIInfo() (*APIInfo, error) {//核心函数,调用api。
    res, err := http.Get(fmt.Sprintf("%s/api-info?key=%s", BaseURL, s.apiKey))//拼接url
    if err != nil {
        return nil, err
    }
    defer res.Body.Close()
​
    var ret APIInfo
    if err := json.NewDecoder(res.Body).Decode(&ret); err != nil {
        return nil, err
    }
    return &ret, nil//返回结构体地址
}

package main
​
import (
    "fmt"
    "log"
    "os"
​
    "github.com/blackhat-go/bhg/ch-3/shodan/shodan"
)
​
func main() {
    if len(os.Args) != 2 {
        log.Fatalln("Usage: main <searchterm>")
    }
    apiKey := os.Getenv("SHODAN_API_KEY")
    s := shodan.New(apiKey)
    info, err := s.APIInfo()
    if err != nil {
        log.Panicln(err)
    }
    fmt.Printf(
        "Query Credits: %d\nScan Credits:  %d\n\n",
        info.QueryCredits,
        info.ScanCredits)
​
    hostSearch, err := s.HostSearch(os.Args[1])
    if err != nil {
        log.Panicln(err)
    }
​
    for _, host := range hostSearch.Matches {
        fmt.Printf("%18s%8d\n", host.IPString, host.Port)
    }
}

​
type Client struct {
    fofa_email string
    fofa_key string
    fofa_query string
}
​
func New(fofa_email string, fofa_key string, fofa_query string) *Client {
    return &Client{fofa_email: fofa_email, fofa_key: fofa_key, fofa_query: fofa_query}
}
​
type APIInfo struct {
    Results [][]string `results`
}
​
func (s *Client) APIInfo() (*APIInfo, error) {
    res, err := http.Get(fmt.Sprintf("%s/api/v1/search/all?email=%s&key=%s&qbase64=%s", BaseURL,s.fofa_email, s.fofa_key,s.fofa_query))
    if err != nil {
        return nil, err
    }
    defer res.Body.Close()
​
    var ret APIInfo
    if err := json.NewDecoder(res.Body).Decode(&ret); err != nil {
        return nil, err
    }
    return &ret, nil
}
